<script>
import ElInput from '@/components/elElements/ElInput.vue'
 
import adminHandler from '@/services/adminHandler'
 
const log = require('script/log')('admin')
 
export default {
  components: { ElInput },
  props: {
    changePassword: {
      default: false,
      type: Boolean
    },
    logs: {
      default: false,
      type: Boolean
    }
  },
  data () {
    return {
      terminals: [{id: 0, input: '', response: ''}],
      terminal: {id: 0, input: '', response: ''},
      currentTerminal: {id: 0, input: '', response: ''},
      flags: {
        sentCmd: 'none'
      },
      selected: 0
    }
  },
  mounted () {
    this.changeWindowFocus(0)
    window.addEventListener('keyup', this.keyUp)
  },
  beforeDestroy: function () {
    window.removeEventListener('keyup', this.keyUp)
  },
  methods: {
    keyUp: function (event) {
      if (event.ctrlKey) {
        for (let i = 49; i < 59; i++) {
          if (event.keyCode === i) {
            this.selectTerminal(i - 49)
          }
        }
 
        switch (event.keyCode) {
          case 68:
            this.sendDb(this.selected)
            break
          case 70:
            this.changeWindowFocus(this.selected)
            break
 
          case 83:
            this.sendCmd(this.selected)
            break
          case 84:
            this.newTerminalWindow()
            break
          case 88:
            this.destroyWindow(this.selected)
            break
        }
      }
    },
 
    getLogs () {
      adminHandler.getLogs()
        .then(files => {
 
        }).catch(err => {
          log.error(err)
        })
    },
    rotateLogs () {
      // todo
    },
    newTerminalWindow () {
      this.terminals.push({id: this.terminals.length, input: '', response: 'Ny terminalruta öppnad!'})
      this.changeWindowFocus(this.terminals.length - 1)
    },
    destroyWindow (index) {
      if (this.terminals.length !== 1) {
        this.selected = index - 1
        if (this.selected < 0) this.selected = 0
        this.terminals.splice(index, 1)
      }
    },
    changeWindowFocus (index) {
      if (index > this.terminals.length - 1) return
      this.currentTerminal = this.terminals[index]
      this.selected = index
      this.$refs.terminalInput.$el.focus()
    },
    selectTerminal (index) {
      this.changeWindowFocus(index)
    },
    sendDb (index) {
      this.setFlags('sentCmd', 'waiting')
      adminHandler.sendDbCmd(this.terminals[index].input)
        .then(response => {
          this.terminals[index].response = response
          this.setFlags('sentCmd', 'success')
        }).catch(err => {
          this.terminals[index].response = err
          log.error(err)
          this.setFlags('sentCmd', 'failed', -1)
        })
    },
    sendCmd (index) {
      this.setFlags('sentCmd', 'waiting')
      adminHandler.sendShellCommand(this.terminals[index].input)
        .then(response => {
          this.terminals[index].response = response
          this.setFlags('sentCmd', 'success')
        }).catch(err => {
          this.terminals[index].response = err
          log.error(err)
          this.setFlags('sentCmd', 'failed', -1)
        })
    },
    sendStreamCmd (index) {
      this.setFlags('sentCmd', 'waiting')
      adminHandler.sendShellStreamCommand(this.shellStreamInput)
        .then(response => {
          this.response = response
          this.setFlags('sentCmd', 'success')
        }).catch(err => {
          this.terminals[index].response = err
          log.error(err)
          this.setFlags('sentCmd', 'failed', -1)
        })
    },
    setFlags (key, value, timer) {
      this.flags[key] = value
      timer = timer || 3000
      if (timer > 0) {
        setTimeout(
          function () {
            this.flags[key] = 'none'
          }.bind(this), timer)
      }
    }
  }
}
