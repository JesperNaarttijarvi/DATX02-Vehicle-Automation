<script>
 
const log = require('@/components/script/log')('H.Bars.')
 
export default {
  name: 'HorizontalSplittedBars',
  props: {
    title: {
      type: String,
      default: ''
    },
    label: {
      type: String,
      default: 'Avsändare'
    },
    unit: {
      type: String,
      default: 'Totalvikt [TON]'
    },
    mean: {
      type: String,
      default: 'Medelvikt [TON]'
    },
    data: {
      type: Array,
      default: () => []
    },
    tooltip: {
      type: Boolean,
      default: true
    }
  },
  data () {
    return {
      globalTooltip: global.tooltip
    }
  },
  mounted () {
    // TODO ADD WATCH FOR GLOBAL.TOOLTIP
    if (global.tooltip === undefined || global.tooltip === null) {
      global.tooltip = true
      this.globalTooltip = true
    }
  },
  computed: {
    totalWeight () {
      let total = 0
      this.data.forEach(item => {
        total += item.value
      })
 
      return Math.round(total * 10) / 10
    },
    totalAbsWeight () {
      let total = 0
      this.data.forEach(item => {
        total += Math.abs(item.value)
      })
 
      return Math.round(total * 10) / 10
    },
    minWeight () {
      let min = 0
      this.data.forEach(item => {
        if (min > item.value) {
          min = item.value
        }
      })
      return Math.round(min * 10) / 10
    },
    maxWeight () {
      let max = 0
      this.data.forEach(item => {
        if (max < item.value) {
          max = item.value
        }
      })
      return Math.round(max * 10) / 10
    }
  },
  methods: {
    getStats (row) {
      return (`${this.unit}: ${(row.value)}, Antal: ${(row.count)}st och ${this.mean}: ${Math.round(row.value / row.count)}`)
    },
    getPercentage (value) {
      // const tot = this.totalWeight
      const totAbs = this.totalAbsWeight
      // const min = this.minWeight
      // const max = this.maxWeight
 
      let percentage = 0
 
      if (totAbs === 0 ||
        typeof totAbs !== 'number' ||
        typeof value !== 'number') {
        log.error('Invalid format')
        return
      }
 
      percentage = value / totAbs
      percentage *= 100
      percentage = Math.round(percentage * 10) / 10
 
      return `${percentage}%`
    }
  }
}
