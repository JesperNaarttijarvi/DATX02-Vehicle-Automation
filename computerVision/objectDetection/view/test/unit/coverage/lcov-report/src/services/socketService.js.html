import io from 'socket.io-client'
import devMode from '@/devMode'
 
const log = require('@/components/script/log')('socket')
 
const socketService = {
  ip: window.location.host,
  port: '',
  socket: '',
  https: false,
  timeout: 5000
}
 
if (devMode.dev && devMode.localServer) {
  socketService.ip = 'localhost:8080'
}
 
socketService.connected = () => true
socketService.testing = () => true
 
socketService.getAddress = (room = '') => {
  let slash = ''
  if (room) slash = '/'
  return `${socketService.ip}${slash}${room}`
}
 
socketService.setSocket = () => {
  if (devMode.dev && !devMode.localServer) {
    log.warning('No socket due to dev mode')
    socketService.socket = {
      emit: (name, args) => {
        log.debug(`Emit ${name}`)
      },
      on: (name, callback) => {
        log.debug(`On listened for ${name}`)
      }
    }
  } else {
    socketService.socket = io(socketService.getAddress())
  }
}
 
socketService.setRoom = (room) => {
  socketService.socket = io(socketService.getAddress(room))
}
 
socketService.getSocket = () => socketService.socket
 
socketService.get = (args, data, _timeout) => new Promise((resolve, reject) => {
  if (args === undefined) return
  let received = false
  let timeout
  let eventName
  let eventData
  let eventRespone
 
  if (data !== undefined) {
    timeout = _timeout || socketService.timeout
    eventName = `get${args}`
    eventData = data || ''
    eventRespone = args
  } else {
    timeout = args.timeout || socketService.timeout
    eventName = `get${args.event}`
    eventData = args.data || ''
    eventRespone = args.event
  }
 
  socketService.request(eventName, eventRespone, eventData)
    .then((data) => {
      if (data.error) {
        reject(data.message)
      } else {
        received = true
        resolve(data)
      }
    })
    .catch(err => reject(err))
 
  setTimeout(() => {
    if (!received) {
      reject(new Error('Timeout when requesting data'))
    }
  }, timeout)
})
 
socketService.set = (args, data, _timeout) => new Promise((resolve, reject) => {
  let received = false
  let timeout
  let eventName
  let eventData
  let eventRespone
 
  if (data !== undefined) {
    timeout = _timeout || socketService.timeout
    eventName = `get${args}`
    eventData = data || ''
    eventRespone = args
  } else {
    timeout = args.timeout || socketService.timeout
    eventName = `get${args.event}`
    eventData = args.data || ''
    eventRespone = args.event
  }
 
  socketService.request(`get${eventName}`, eventRespone, eventData)
    .then((data) => {
      if (data.error) {
        reject(data.message)
      } else {
        received = true
        resolve(data)
      }
    })
    .catch(err => reject(err))
 
  setTimeout(() => {
    if (!received) {
      reject(new Error('Timeout when requesting data'))
    }
  }, timeout)
})
 
socketService.request = (eventName, eventRespone, eventArgs) => new Promise((resolve, reject) => {
  socketService.socket.emit(eventName, eventArgs)
  socketService.socket.on(eventRespone, (data) => {
    if (data.error) {
      reject(data.msg, data.error)
    }
    resolve(data)
  })
})
 
socketService.addListener = (eventName, callback) => {
  socketService.socket.on(eventName, (eventArgument) => {
    callback(eventArgument)
  })
}
 
socketService.setSocket()
 
export default socketService
