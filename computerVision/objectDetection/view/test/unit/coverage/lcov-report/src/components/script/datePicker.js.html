const dates = {}
 
dates.getLatestWeek = () => {
  const date = new Date()
  let day = date.getDay()
 
  if (day === 0) {
    day = 7
  }
 
  const to = new Date(date.setDate(date.getDate() - date.getDay()))
  const from = new Date(date.setDate(to.getDate() - 6))
 
  return { to, from }
}
 
export default dates
