import socket from './socketService'
import observer from '@/components/script/observer'
import devMode from '@/devMode'
// const log = require('@/components/script/log')('dashboardHandler.')
 
const dashboardHandler = {
  connect: true
}
 
if (devMode.localServer || dashboardHandler.connect) {
  socket.setRoom('kaaDashboard')
  socket.addListener('dashboardData', newDashboardData)
}
 
function newDashboardData (values) {
  /* EXPECTED KEYS
    lagerInOut / outgoingTransports / incomingTransports / nowTransports / lastTransports / vvBuffer
    If value[key] is undefined, there is no new data in it
  */
  for (let key in values) {
    if (values[key] !== undefined) observer.emit(key, values[key])
  }
}
 
export default dashboardHandler
