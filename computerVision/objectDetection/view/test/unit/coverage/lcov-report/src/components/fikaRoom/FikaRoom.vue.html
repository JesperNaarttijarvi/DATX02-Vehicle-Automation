<script>
 
import HorizontalBars from '@/components/elements/HorizontalBars.vue'
import TrafficTable from '@/components/elements/TrafficTable.vue'
import ProgressBar from '@/components/elements/ProgressBar.vue'
import ElInput from '@/components/elElements/ElInput.vue'
 
import visualJS from '@/modules/visualJS.min.js'
import dashboardHandler from '@/services/kaaDashboardHandler'
import observer from '@/components/script/observer'
 
let diagramInOut = null
let diagramData = null
 
export default {
  name: 'fikaRoom',
  components: { HorizontalBars, TrafficTable, ProgressBar, visualJS, ElInput },
  data () {
    return {
      nrOfDaysInOut: '9',
      outgoingTransports: [],
      incomingTransports: [],
      nowTransports: [],
      lastTransports: [],
      buffertPercentage: 0,
      buffertMean: 0,
      isIncoming: false,
      isNow: false
    }
  },
  mounted () {
    observer.listen('outgoingTransports', this.setOutgoingTransports)
    observer.listen('lagerInOut', this.setLagerInOut)
    observer.listen('incomingTransports', this.setIncomingTransports)
    observer.listen('nowTransports', this.setNowTransports)
    observer.listen('lastTransports', this.setLastTransports)
    observer.listen('vvBuffer', this.setVvBuffer)
    observer.listen('resize', this.updateDiagram)
 
    setTimeout(() => {
      observer.emit('resize')
    }, 2000)
  },
  methods: {
    adjustDays (int) {
      let days = Number(this.nrOfDaysInOut) + int
      if (days < 5) days = 5
      this.nrOfDaysInOut = '' + days
    },
    setVvBuffer (args) {
      if (!args) return
 
      const mean = Number(args.mean) || 0
      const current = Number(args.current) || 0
      let percentage = 0
 
      if (typeof mean === 'number' && mean > 0) {
        percentage = current / mean
      }
 
      this.buffertPercentage = percentage * 100
      this.buffertMean = Number(mean)
    },
    getOutgoingTransports () {
      dashboardHandler.getOutgoing()
        .then(data => {
          this.setOutgoingTransports(data)
        }).catch(err => {
          console.log(err)
          global.bugsnagClient.notify(err)
        })
    },
    setOutgoingTransports (data) {
      this.outgoingTransports = data.map(item => ({
        value: Math.round(item.sum),
        label: item.articleName,
        id: item.article,
        color: item.articleColor
      }))
 
      observer.emit('resize')
    },
    setLagerInOut (rawData) {
      console.log(rawData)
      const dates = []
      const outValues = []
      const inValues = []
 
      rawData.forEach(dataPoint => {
        dates.push(new Date(dataPoint.date))
        outValues.push(dataPoint.outKaa)
        inValues.push(dataPoint.inKaa)
      })
 
      const data = {
        data: {
          1: {
            title: 'Inkommande material',
            x: dates,
            y: inValues,
            type: 'line',
            themeColor: 'red',
            thickness: 5,
            opacity: 0.85
          },
          2: {
            title: 'Utgående material',
            x: dates,
            y: outValues,
            type: 'line',
            themeColor: 'green',
            thickness: 5,
            opacity: 0.85
          }
        },
        margin: {
          top: 20,
          left: 20,
          right: 20,
          bottom: 20
        },
        xAxis: {
          title: 'Datum',
          markers: 5,
          grid: false,
          size: 60,
          textMarkFontSize: 16,
          captionSpacing: 10,
          type: 'date',
          textMarkMargin: 6
        },
        xAxisInverted: {
          markers: 5,
          type: 'date',
          title: 'Datum',
          textMarkFontSize: 16,
          grid: false,
          size: 60,
          textMarkMargin: 10
        },
        yAxis: {
          title: 'Ton',
          markers: 6,
          subMarkers: 2,
          size: 80,
          textMarkFontSize: 16,
          min: 0,
          max: 300,
          type: 'number'
        },
        yAxisInverted: {
          size: 70,
          markers: 6,
          subMarkers: 2,
          textMarkFontSize: 16,
          grid: true,
          type: 'number'
        },
        guideLines: {
          x: true,
          y: true
        },
        legend: {
          fontSize: 16,
          size: 10,
          margin: 10,
          spaceBetweenDotAndLabel: 5,
          position: 'top'
        },
        header: false,
        fontFamily: 'Arial',
        toolTip: true,
        toolTipClosest: true,
        debug: false,
        caption: {
          caption: 'Inkommande material till och från KAA. Diagrammet visar summerat TRÄ, BRÄNNBART och HUSHÅLLSAVFALL',
          size: 10,
          align: 'left',
          styling: {}
        }
      }
 
      diagramData = Object.assign({}, data)
 
      let container = document.getElementById('lineDiagramId')
 
      while (container.firstChild) {
        container.removeChild(container.firstChild)
      }
 
      diagramInOut = visualJS.new(container, data)
    },
    setIncomingTransports (data) {
      this.incomingTransports = data
      this.isIncoming = data.length > 0
    },
    setNowTransports (data) {
      this.nowTransports = data
      this.isNow = data.length > 0
    },
    setLastTransports (data) {
      this.lastTransports = data
    },
    updateDiagram () {
      if (diagramInOut === null || diagramData === null) return
 
      const data = Object.assign({}, diagramData)
 
      diagramInOut = visualJS.update(diagramInOut, data)
    }
  }
