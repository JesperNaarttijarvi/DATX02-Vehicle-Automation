<template>
<card-tile>
  <div ref="VisualJSCont" class="visualJS">
  </div>
</card-tile>
</template>
<script>
 
import visualJS from '@/modules/visualJS.min.js'
 
export default {
  name: 'VisualJS',
  props: {
    data: {
      type: Object,
      default: () => ({
        title: 'Visual JS Default template',
        visualJS: {},
        diagram: {}
      })
    }
  },
  mounted () {
    const diagramData = this.data.visualJS
    const diagramCont = this.$refs.VisualJSCont
    // Initialize Visual JS Diagram'
    this.data.diagram = visualJS.new(diagramCont, JSON.parse(JSON.stringify(diagramData)))
    window.addEventListener('resize', this.updateWidth)
    // TODO, FIX RESIZE ON RESCALE
  },
  methods: {
    updateWidth () {
      const diagramData = this.data.visualJS
      const diagramCont = this.$refs.VisualJSCont
      this.data.diagram.clear()
      diagramCont.innerHTML = ''
      // Initialize Visual JS Diagram'
      this.data.diagram = visualJS.new(diagramCont, JSON.parse(JSON.stringify(diagramData)))
    }
  }
}
</script>
