import axios from 'axios'
import devMode from '@/devMode'
import qs from 'qs'
 
let baseURL = window.location.host
const httpService = {
  // baseURL: window.location.host,
  headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
  //  headers: { 'Content-Type': 'application/json' },
  timeout: 60000,
  totalTimeout: 300000,
  maxRetries: 5,
  retry: true
}
 
if (devMode.dev) {
  httpService.baseURL = 'http://localhost:8080'
  baseURL = 'http://localhost:8080'
}
 
httpService.get = (args) => new Promise((resolve, reject) => {
  if (typeof args === 'string') {
    let url = args
    args = { url }
  }
  args.method = 'get'
  httpService.request(args)
    .then((data) => {
      resolve(data)
    })
    .catch(err => reject(err))
})
 
httpService.post = (args) => new Promise((resolve, reject) => {
  if (typeof args === 'string') {
    let url = args
    args = { url }
  }
  args.method = 'post'
  args.data = args.body || args.data || ''
 
  httpService.request(args)
    .then((data) => {
      resolve(data)
    })
    .catch(err => reject(err))
})
 
httpService.request = (args) => new Promise((resolve, reject) => {
  let received = false
 
  const cbResolve = (data) => {
    received = true
    resolve(data)
  }
  const cbReject = (err) => {
    reject(err)
  }
 
  if (args.retry === undefined) args.retry = httpService.retry
 
  args.nrOfAttempts = args.nrOfAttempts || 0
  args.maxRetries = args.maxRetries || httpService.maxRetries
  // args.baseURL = args.baseURL || httpService.baseURL
  args.baseURL = ''
  args.timeout = args.timeout || httpService.timeout
  args.totalTimeout = args.totalTimeout || httpService.totalTimeout
  args.headers = args.headers || httpService.headers
  args.data = args.data || ''
 
  if (!devMode.dev) {
    args.url = `https://${baseURL}/${args.url}`
  } else {
    args.url = `${baseURL}/${args.url}`
  }
 
  if (args.headers['Content-Type'] !== 'application/json') {
    args.data = qs.stringify(args.data)
  }
 
  tryRequesting(args, cbResolve, cbReject)
 
  setTimeout(() => {
    if (!received) {
      reject(new Error('Timeout when requesting data'))
    }
  }, args.totalTimeout)
})
 
const tryRequesting = (args, cbResolve, cbReject) => {
  args.nrOfAttempts++
  // Method defaults to get
  axios(args)
    .then((data) => {
      cbResolve(data)
    })
    .catch(err => {
      if (args.nrOfAttempts > args.maxRetries || !args.retry) {
        cbReject(err)
        return
      } else {
        global.bugsnagClient.notify(err)
      }
      setTimeout(() => {
        tryRequesting(args, cbResolve, cbReject)
      }, 200 * args.nrOfAttempts * args.nrOfAttempts)
    })
}
 
export default httpService
