import httpService from '@/services/httpService'
 
const facility = {
  baseUrl: 'facilities/'
}
 
facility.oljehallExternalWebsite = () => new Promise((resolve, reject) => {
  httpService.get(`${facility.baseUrl}oljehallExternalWebsite`)
    .then((response) =>
      resolve(response)
    ).catch(err => reject(err))
})
 
facility.getOljehallEvents = () => new Promise((resolve, reject) => {
  httpService.get(`${facility.baseUrl}getOljehallEvents`)
    .then((response) => {
      const larms = {
        A: {
          day: 0,
          week: 0,
          month: 0,
          quarter: 0,
          year: 0
        },
        B: {
          day: 0,
          week: 0,
          month: 0,
          quarter: 0,
          year: 0
        },
        C: {
          day: 0,
          week: 0,
          month: 0,
          quarter: 0,
          year: 0
        }
      }
 
      const addEvent = (type, str) => {
        larms[type][str] += 1
      }
 
      const date = new Date()
      date.setDate(date.getDate() - 1)
      const day = parseDate(date)
      date.setDate(date.getDate() - 6)
      const week = parseDate(date)
      date.setDate(date.getDate() - 24)
      const month = parseDate(date)
      date.setDate(date.getDate() - 66)
      const quarter = parseDate(date)
 
      response.data.forEach(event => {
        if (event.date > day) {
          addEvent(event.prio, 'day')
        }
        if (event.date > week) {
          addEvent(event.prio, 'week')
        }
        if (event.date > month) {
          addEvent(event.prio, 'month')
        }
        if (event.date > quarter) {
          addEvent(event.prio, 'quarter')
        }
 
        addEvent(event.prio, 'year')
      })
 
      resolve({events: response.data, larms})
    }).catch(err => reject(err))
})
 
facility.setOljehallFilters = (userFilter) => {
  let facilities = JSON.parse(localStorage.getItem('facilities'))
 
  if (!(facilities instanceof Object)) {
    facilities = {}
  }
 
  if (!facilities.oljehall) {
    facilities.oljehall = {}
  }
 
  facilities.oljehall.filter = userFilter
 
  localStorage.setItem('facilities', JSON.stringify(facilities))
}
 
facility.getOljehallFilters = () => new Promise((resolve, reject) => {
  const filters = JSON.parse(localStorage.getItem('facilities'))
 
  if (filters instanceof Object) {
    if (filters.oljehall) {
      if (filters.oljehall.filter instanceof Object) {
        resolve(filters.oljehall.filter)
        return
      }
    }
  }
 
  resolve({
    prioA: true,
    prioB: true,
    prioC: true,
    activated: true,
    switched: true,
    other: true
  })
})
 
const parseDate = (date) => {
  if (!(date instanceof Date)) date = new Date(date)
 
  let year = date.getFullYear() + ''
  let month = date.getMonth() + 1 + ''
  let day = date.getDate() + ''
  let hours = date.getHours() + ''
  let minutes = date.getMinutes() + ''
  let seconds = date.getSeconds() + ''
 
  if (month.length === 1) month = '0' + month
  if (day.length === 1) day = '0' + day
  if (hours.length === 1) hours = '0' + hours
  if (minutes.length === 1) minutes = '0' + minutes
  if (seconds.length === 1) seconds = '0' + seconds
 
  return `${year}-${month}-${day} ${hours}:${minutes}:${seconds}`
}
 
export default facility
