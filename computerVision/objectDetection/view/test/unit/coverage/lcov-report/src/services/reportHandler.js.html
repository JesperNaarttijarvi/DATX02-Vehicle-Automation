// import socket from './socketService'
import httpService from '@/services/httpService'
 
// const log = require('@/components/script/log')('reportHandler')
 
const report = {
  baseUrl: 'report/'
}
 
// FIXME: Shouldn't this be a get-request instead of post?
report.getAvailableReports = () => new Promise((resolve, reject) => {
  const url = `${report.baseUrl}getAvailableReports`
 
  httpService.get({ url })
    .then(response => {
      const reports = []
      for (let key in response.data) {
        const report = response.data[key]
        reports.push({id: report.id, label: capitalize(report.label), filter: report.filter})
      }
      resolve(reports)
    })
    .catch(err => reject(err))
})
 
report.getSuggestedIntervals = () => new Promise((resolve, reject) => {
  const url = `${report.baseUrl}getSuggestedIntervals`
 
  httpService.get(url)
    .then(response => {
      const data = response.data
      resolve(data)
    })
    .catch(err => reject(err))
})
 
report.getReportData = (reportId, startDate, endDate, filter) => new Promise((resolve, reject) => {
  if (startDate === null) return
  if (endDate === null) return
 
  const params = { reportId: (reportId), filterId: (filter), startDate: (startDate), endDate: (endDate) }
  const url = `${report.baseUrl}getReportData`
 
  httpService.get({ url, params })
    .then(response => {
      const data = response.data
      resolve(data)
    })
    .catch(err => reject(err))
})
 
const capitalize = (string) => {
  string = string + ''
  if (string.length < 1) return string
  const firstLetter = string.charAt(0).toUpperCase()
  const stringWithoutFirstLetter = string.substring(1)
  return firstLetter + stringWithoutFirstLetter
}
 
export default report
