<script>
 
const log = require('@/components/script/log')('DeviationPage')
 
export default {
  name: 'reportPage',
  components: { },
  data () {
    return {
      filter: {
        options: [],
        model: null,
        selectedFilter: ''
      },
      report: {
        id: -1,
        from: null,
        to: null,
        filter: null
      },
      loader: {
        success: false,
        error: false,
        isLoading: false,
        caption: ''
      }
    }
  },
  props: ['reportId', 'fromDate', 'toDate', 'filterId'],
  mounted () {
    let okProps = this.okProps()
 
    if (okProps) {
      this.getDefaultDataWithProps()
    } else {
      this.getDefaultData()
    }
  },
  methods: {
    formatDate (date) {
      let d = date.getDate()
      let m = date.getMonth() + 1
      const y = date.getFullYear()
 
      if (d < 10) d = `0${d}`
      if (m < 10) m = `0${m}`
 
      return `${y}-${m}-${d}`
    },
    chooseReport (report) {
      this.selectedReport = `${report.label} ${this.filter.selectedFilter}`
      this.selectedReportId = report.id
 
      this.updateFilter()
      this.getReportData()
    },
    updateFilter () {
      log.debug('update filter')
      this.filter.options = []
      this.filter.model = null
 
      this.availableReports.forEach(report => {
        if (
          report.id === this.selectedReportId &&
          report.filter !== undefined
        ) {
          this.filter.options = report.filter
          this.filter.model = report.filter[0].value
        }
      })
    },
    chooseInterval (interval) {
      this.timeInterval = interval.label
      this.selectedDate = interval.value
      this.timeIntervalHeader = interval.header
      this.customDateSelected = false
      this.getReportData()
    },
    selectCustomDate () {
      this.selectedDate.from = this.formatDate(this.customDate.from)
      this.selectedDate.to = this.formatDate(this.customDate.to)
 
      this.showDateSelector = false
      this.customDateSelected = true
 
      this.timeIntervalHeader = `Tidsintervall: ${this.selectedDate.from} - ${
        this.selectedDate.to
      }`
      this.getReportData()
    },
    okProps () {
      if (
        this.reportId === undefined ||
        typeof this.fromDate !== 'string' ||
        typeof this.toDate !== 'string'
      ) {
        return false
      }
 
      return true
    },
    async getDefaultData () {
      log.debug('Get default data')
 
      await this.getAvailableReports()
      await this.getSuggestedIntervals()
      // log.todo('setTimeOut when getting reportData. Fix the async wait')
      // setTimeout(this.getReportData, 150)
 
      this.getReportData()
    },
    async getDefaultDataWithProps () {
      log.debug('Get default data with props')
 
      await this.getAvailableReports()
      await this.getSuggestedIntervals()
 
      /*
      // Run them in parallell
      await Promise.all([
        this.getAvailableReports(),
        this.getSuggestedIntervals()
      ])
 
      // Catch err
      try {
        await Promise.all([
          this.getAvailableReports(),
          this.getSuggestedIntervals()
        ])
 
        // throw 'something went wrong'
      } catch(err) {
        log.error('Could not initialize', err)
      }
      */
 
      log.debug('Set props data')
 
      this.selectedReportId = Number(this.reportId)
      this.selectedReport = 'Inväntar data...'
      this.selectedDate = {
        from: this.fromDate,
        to: this.toDate
      }
 
      this.availableReports.forEach(report => {
        if (report.id === this.selectedReportId) {
          this.selectedReport = report.label
        }
      })
 
      this.getReportData()
    },
    async getAvailableReports () {
      log.debug('Get available reports')
 
      return reportHandler
        .getAvailableReports()
        .then(reports => {
          this.availableReports = reports
 
          this.selectedReport = reports[0].label
          this.selectedReportId = reports[0].id
          this.updateFilter()
 
          log.success('Available reports response')
 
          return Promise.resolve()
        })
        .catch(err => {
          log.error('Could not get available reports', err)
        })
    },
    async getSuggestedIntervals () {
      log.debug('Get suggested intervals')
 
      return reportHandler
        .getSuggestedIntervals()
        .then(intervals => {
          // intervals = [{ label, header, value: {from, to}}]
          this.suggestedIntervals = intervals
 
          this.timeIntervalHeader = intervals[0].header
          this.timeInterval = intervals[0].label
          this.selectedDate = intervals[0].value
 
          log.success('Suggested intervals response')
 
          return Promise.resolve()
        })
        .catch(err => {
          log.error('Could not get intervals', err)
        })
    },
    getReportData () {
      this.loader.isLoading = true
      log.debug(`Get report data for id: ${this.selectedReportId}`)
      this.reportContent = {}
      reportHandler
        .getReportData(
          this.selectedReportId,
          this.selectedDate.from,
          this.selectedDate.to,
          this.filter.model
        )
        .then(data => {
          this.reportContent = data
 
          this.resetLoader()
 
          log.success('Report data response')
        })
        .catch(err => {
          this.loader.error = true
          this.loader.caption = 'Kunde inte hämta rapporten.'
          log.error('Could not get report data', err)
        })
    },
    setData (data) {
      // Translate articles to data
      this.data = data.map(item => ({
        value: Math.round(item.sum),
        label: item.articleName,
        id: item.article,
        color: item.articleColor
      }))
    },
    resetLoader () {
      this.loader.success = false
      this.loader.error = false
      this.loader.isLoading = false
      this.loader.caption = ''
    }
  }
}
