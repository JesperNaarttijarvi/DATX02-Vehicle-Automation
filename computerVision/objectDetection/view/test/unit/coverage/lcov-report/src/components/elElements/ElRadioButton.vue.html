<template>
<div>
  <label
    v-for="(item, index) in options"
    class="radioContainer"
    :class="{ marginTop, marginRight }"
    :key="index"
    :value="value"
    >
 
    {{item.label}}
 
    <input
      type="radio"
 
      :id="index"
      :value="item.value"
      v-model="checked"
 
      @click="onChange(item.value)"
      >
 
    <span class="checkmark" :value="value"></span>
  </label>
</div>
</template>
 
<script>
 
export default {
  name: 'ElRadioButton',
  data () {
    return {
      checked: null
    }
  },
  props: {
    value: {
      type: String,
      default: ''
    },
    options: {
      type: Array,
      default: () => []
    },
    marginTop: {
      type: Boolean,
      default: false
    },
    marginRight: {
      type: Boolean,
      default: false
    }
  },
  mounted () {
    this.checked = this.value
  },
  watch: {
    value () {
      this.checked = this.value
    }
  },
  methods: {
    onChange (value) {
      this.$emit('input', value)
      this.$emit('click', value)
    }
  }
}
</script>
