user.getAllNotifications = () => new Promise((resolve, reject) => {
  httpService.get(`${user.baseUrl}getAllNotifications`)
    .then(records => {
      const notifications = []
      records.data.forEach(not => {
        notifications.push({
          id: Number(not.id),
          name: not.name,
          description: not.description,
          enable: not.enable === 'true',
          subgroup: Number(not.subgroup),
          subgroupName: (not.subgroupName),
          type: Number(1)
        })
      })
 
      const sortedNots = {}
      notifications.forEach(not => {
        const groupName = not.subgroupName
        if (sortedNots[groupName]) {
          sortedNots[groupName].push(not)
        } else {
          sortedNots[groupName] = [not]
        }
      })
      resolve(sortedNots)
    }).catch(err => reject(err))
})
 
user.saveNotifications = (groupedNots) => new Promise((resolve, reject) => {
  const keys = Object.keys(groupedNots)
  const notifications = []
  keys.forEach(key => {
    groupedNots[key].forEach(not => {
      notifications.push(not)
    })
  })
 
  const params = {
    url: `${user.baseUrl}setUserNotifications`,
    data: { 'notifications': notifications }
  }
 
  httpService.post(params)
    .then(() => resolve())
    .catch(err => reject(err))
})

user.changePassword = ({ oldPassword, newPassword }) => new Promise((resolve, reject) => {
  if (newPassword === '') {
    reject(new Error('Empty password'))
    return
  }
 
  const params = {
    url: `${user.baseUrl}changePassword`,
    data: {
      'oldPassword': oldPassword,
      'newPassword': newPassword
    },
    retry: false
  }
 
  httpService.post(params)
    .then(records => {
      resolve(records)
    }).catch(err => {
      console.log(err)
      reject(err)
    })
})

user.getPendingDeviationFilters = () => new Promise((resolve, reject) => {

    .then(response => {
      const filter = {}
      filter.filter = response.data.filter
      filter.options = []
      const filterOptions = response.data.options
 
      for (let i = 0; i < filterOptions.length; i++) {
        filter.options[i] =
          {
            value: Number(filterOptions[i].value),
            label: `${filterOptions[i].label} (${filterOptions[i].value})`
          }
      }
      resolve(filter)
    })
    .catch(err => reject(err))
})
