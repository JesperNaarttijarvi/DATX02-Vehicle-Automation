<template>
<div class="selectContainer"
  v-bind:class="{
    marginRight,
    marginTop,
    wide
  }"
  >
  <select @change="onChange"
    class="selectContainer-field"
    :value="value"
    >
 
    <option v-for="item in options"
      :key="item.value"
      :value="item.value"
      :disabled="item.value === null"
      >
      {{item.label}}
    </option>
  </select>
</div>
</template>
 
<script>
 
export default {
  name: 'ElDropdown',
  props: {
    value: {
      default: null
    },
    options: {
      type: Array,
      default: () => []
    },
    marginRight: {
      type: Boolean,
      default: false
    },
    marginTop: {
      type: Boolean,
      default: true
    },
    wide: {
      type: Boolean,
      default: false
    }
  },
  methods: {
    onChange (ev) {
      this.$emit('input', ev.target.value)
    }
  }
}
 
</script>
